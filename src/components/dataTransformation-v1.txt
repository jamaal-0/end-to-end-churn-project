import numpy as np
import pandas as pd

from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import (
    OneHotEncoder,
    PowerTransformer,
    RobustScaler,
    StandardScaler
)


class dataTransformation:

    def __init__(self, targetColumn: str):
        self.targetColumn = targetColumn

    # ------------------------------
    # Preprocessor Builder
    # ------------------------------
    def preprocess(
        self,
        normalNumericCols,
        skewedNumericCols,
        outliersNumericCols,
        skew_outlier_cols,
        normalCategoricalCols,
        imbalancedcategoricalCols
    ):

        transformers = []

        # ---- Numeric pipelines ----

        if normalNumericCols:
            normalNumericPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="mean")),
                ("scaler", StandardScaler())
            ])

            transformers.append(
                ("num_normal", normalNumericPipeline, normalNumericCols)
            )

        if skewedNumericCols:
            skewedNumericPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="median")),
                ("power", PowerTransformer(method="yeo-johnson")),
                ("scaler", StandardScaler())
            ])

            transformers.append(
                ("num_skewed", skewedNumericPipeline, skewedNumericCols)
            )

        if outliersNumericCols:
            outlierPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="median")),
                ("scaler", RobustScaler())
            ])

            transformers.append(
                ("num_outliers", outlierPipeline, outliersNumericCols)
            )

        if skew_outlier_cols:
            skewOutlierPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="median")),
                ("power", PowerTransformer(method="yeo-johnson")),
                ("scaler", RobustScaler())
            ])

            transformers.append(
                ("num_skew_outlier", skewOutlierPipeline, skew_outlier_cols)
            )

        # ---- Categorical pipelines ----

        if normalCategoricalCols:
            normalCatPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="most_frequent")),
                ("encoder", OneHotEncoder(
                    handle_unknown="ignore",
                    sparse_output=False
                ))
            ])

            transformers.append(
                ("cat_normal", normalCatPipeline, normalCategoricalCols)
            )

        if imbalancedcategoricalCols:
            imbalancedCatPipeline = Pipeline([
                ("imputer", SimpleImputer(strategy="most_frequent")),
                ("encoder", OneHotEncoder(
                    handle_unknown="infrequent_if_exist",
                    min_frequency=0.01
                ))
            ])

            transformers.append(
                ("cat_imbalanced", imbalancedCatPipeline, imbalancedcategoricalCols)
            )

        return ColumnTransformer(transformers)

    # ------------------------------
    # Data Transformation Pipeline
    # ------------------------------
    def transformData(self, trainPath, testPath, schema):

        train_df = pd.read_csv(trainPath)
        test_df = pd.read_csv(testPath)

        train_df = train_df.copy()
        test_df = test_df.copy()

        # ---- Schema unpack ----

        normalNumericCols = schema["normalNumericCols"]
        skewedNumericCols = schema["skewedNumericCols"]
        outliersNumericCols = schema["outliersNumericCols"]
        skew_outlier_cols = schema["skew_outlier_cols"]
        normalCategoricalCols = schema["normalCategoricalCols"]
        imbalancedcategoricalCols = schema["imbalancedcategoricalCols"]
        drop_columns = schema["drop_columns"]

        # ---- Separate target ----

        y_train = train_df[self.targetColumn].map({'No': 0, 'Yes': 1}).values
        y_test = test_df[self.targetColumn].map({'No': 0, 'Yes': 1}).values

        train_df = train_df.drop(columns=[self.targetColumn], errors="ignore")
        test_df = test_df.drop(columns=[self.targetColumn], errors="ignore")

        # ---- Drop schema columns ----

        train_df = train_df.drop(columns=drop_columns, errors="ignore")
        test_df = test_df.drop(columns=drop_columns, errors="ignore")

        # ---- Numeric coercion ----

        for df in [train_df, test_df]:
            if "TotalCharges" in df.columns:
                df["TotalCharges"] = pd.to_numeric(
                    df["TotalCharges"],
                    errors="coerce"
                )

        # ---- Build pipeline ----

        preprocessor = self.preprocess(
            normalNumericCols,
            skewedNumericCols,
            outliersNumericCols,
            skew_outlier_cols,
            normalCategoricalCols,
            imbalancedcategoricalCols
        )

        # ---- Fit transform training data ----

        X_train = preprocessor.fit_transform(train_df)
        X_test = preprocessor.transform(test_df)

        # ---- Ensure numpy array safety ----

        #X_train = np.atleast_2d(X_train)
        #X_test = np.atleast_2d(X_test)

        print(X_train, '\n',y_train, '\n',X_test, '\n',y_test,'\n', end='\n')
        

        return X_train, y_train, X_test, y_test, preprocessor
